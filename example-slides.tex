    use, intrinsic :: iso_fortran_env, only: output_unit, &
                                             input_unit
    implicit none
    character(len=1024) :: who
    write(output_unit, *) "Who is this?"
    read(input_unit, *) who
    write(output_unit, *) "Hello ", trim(who)
end program hello
